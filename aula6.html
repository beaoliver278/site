Código C# para o circuito:
using System;
using System.Threading.Tasks;

namespace CircuitoBotao
{
    class Program
    {
        static async Task Main(string[] args)
        {
            Console.WriteLine("Pressione o botão para acender o LED.");
            while (true)
            {
                if (Console.ReadKey().Key == ConsoleKey.Enter)
                {
                    Console.WriteLine("LED aceso!");
                }
                await Task.Delay(100);
            }
        }
    }
}
